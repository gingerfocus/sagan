#+begin_src matlab

Vs = 5;
Rs = 1e5;
Rvm = 1e6;

R1 = 12e3;

V1 = (R1 * Vs) / (Rs + R1)
Req = (R1 * Rvm) / (R1 + Rvm);
V1L = Vs * ( (R1 * Rvm) / ( Rs * (R1 + Rvm) + R1 + Rvm ) )
E1 = (V1L - V1) / V1 * 100
RvmL = ( (Rs + 1) * R1 * V1L ) / ( (Vs * R1) - (Rs * V1L) - V1L )
#+end_src
