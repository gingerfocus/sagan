#+BEGIN_SRC asm
        add     x18, x0, x0
        addi    x29, x0, 100
        addi    x30, x0, 20
Loop:   lw      x7, 0(x10)          ; dereference pointer
        add     x19, x19, x7
        bge     x7, x30, End
        addi    x10, x10, 4         ; increment to next item
        addi    x18, x18, 1         ; increment index
        blt     x18, x29, Loop
End:                                ; pointer still in x10 position
#+END_SRC

The above assembly code can be rewriten into the following c code block:

#+BEGIN_SRC c
int *p1raw(int *array) { // x10
    int i = 0; // add x18, x0, x0
    int result, p; // x19, x7
    int x29 = 100;
    int x30 = 20;

loop:
    p = *array;             // lw  x7, 0(x10)
    result += p;            // add x19, x19, x7
    if (p > x30) break;     // bge x7, x30, End
    array = &array[1];      // addi x10, x10, 4
    i += 1;                 // addi x18, x18, 1
    if (i < x29) goto loop;

    return array;         // at end of block value in r10 is the array pointer
}
#+END_SRC

Or further cleaned up into more human readable code:

#+BEGIN_SRC c
int *p1clean(int *array) {
    int i, p, result;

    for (i = 0; i < 100; i++) { 
        p = *array;
        result += p;
        if (p >= 20) break;
        array++;
    }
    return array;
}
#+END_SRC
